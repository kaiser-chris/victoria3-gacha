sgui_gacha_activate = {
    scope = country

    is_shown = {
        has_variable = gacha_selected
    }

    effect = {
        remove_variable = gacha_selected
    }
}

sgui_gacha_activate_one = {
    scope = country

    effect = {
        activate_gacha_ball = {
            slot = gacha_active_one
        }
    }
}
