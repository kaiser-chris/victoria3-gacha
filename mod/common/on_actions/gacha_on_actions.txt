# Root = any
on_monthly_pulse = {
    on_actions = {
        on_monthly_gacha_ui_check
        on_create_gacha_types
    }
}

# Root = any
on_yearly_pulse_country = {
    on_actions = {
        on_add_gacha_free_pulls
    }
}

on_monthly_gacha_ui_check = {
    effect = {
        add_gacha_main_button = yes
    }
}

on_game_started = {
    on_actions = {
        on_create_gacha_types
    }
}

on_create_gacha_types = {
    effect = {
        create_gacha_types = yes
    }
}

on_add_gacha_free_pulls = {
    trigger = {
        is_player = yes
    }
    effect = {
        add_gacha_free_pulls = {
            amount = 1
        }
    }
}
